## Event

제이쿼리의 이벤트를 연결하는 방식으로 자주쓰는 두종류가 있다.

1. 기본식
   => 이벤트를 바인딩 할 수 있다.(하나의 요소에 여러 이벤트를 사용할 수 있다.)

```js
// $('').on('이벤트명 이벤트명...', 콜백함수)
$(".bu1").on("click", function () {
  $(this).tex("버튼클릭함");
});

// on을 사용한 바인딩 최신 방법
$(".p1").on("mouseenter mouseleave", function () {
  $(this).after("다시 빠져나간다");
});

// on을 사용해서 여러 이벤트를 , 으로 묶음
$(".p1").on({
  click: function () {
    $(this).after("클릭했다.");
  },
  mouseenter: function () {
    $(this).after("들어왔다.");
  },
  mouseleave: function () {
    $(this).after("나갔다.");
  },
});
```

```html
<head>
  <script src="https://code.jquery.com/jquery-3.7.1.min.js"></script>
</head>
<body>
  <form>
    <select id="sel">
      <option value="">선택</option>
      <option value="pizza">pizza</option>
      <option value="pasta">pasta</option>
      <option value="steak">steak</option>
    </select>
    <p>음식 : <span class="result"></span></p>
  </form>
</body>
<script>
  // 자바스크립트로
  let n = document.getElementById("sel");

  n.addEventListener("change", function (e) {
    let r = document.querySelector(".result");
    r.textContent = `${e.target.value}`;
  });

  // 제이쿼리로
</script>
```
