### scroll

```html
<!DOCTYPE html>
<html lang="en">
  <head>
    <script src="https://code.jquery.com/jquery-3.7.1.min.js"></script>
    <style>
      body {
        width: 200vw;
        height: 200vh;
      }
      .di {
        position: fixed;
        border: 1px solid blue;
      }
    </style>
  </head>
  <body>
    <div class="di">
      <h3>스크롤</h3>
      <span class="s1"></span>
      <!-- 스크롤 위치를 표시할 요소 -->
    </div>
    <script>
      $(window).scroll(function () {
        let left = $(this).scrollLeft(); // 왼쪽위치 가져오기
        let top = $(this).scrollTop(); //상단위치 가져오기
        $(".s1").text(left + " , " + top);
      });
    </script>
  </body>
</html>
```
